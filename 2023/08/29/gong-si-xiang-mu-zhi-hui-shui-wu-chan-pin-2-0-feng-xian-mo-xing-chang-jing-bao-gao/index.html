        select scene_id,scene_name from(
            select s.scene_id, s.scene_name , count(r.scene_id) as count

            order by count desc

@Override
    public PageDomain<RcModelDomain> selectPageByEntity(RcModelDomain paramEntity, PageParam pageParam) {
        return this.selectPageByEntityWrapper(paramEntity, pageParam, (p) -> {
            //根据查询条件查出数据
            List<RcModelDomain> resultList = this.mapper.selectByEntityLike(p);
            //将一些字段从数字转为中文
            HashMap<Integer,String> mapOfDictData=new HashMap<>();
            List<DictDataDomain> modelStatusList = GlobalDictUtils.getDataList("modelStatus");
            for(int i=0;i<modelStatusList.size();i++){
                mapOfDictData.put(Integer.parseInt(modelStatusList.get(i).getDictValue()),modelStatusList.get(i).getDictLabel());
            }
            // 存储的为  1 -> 模型A     2->模型B

            for(int i=0;i<resultList.size();i++){
                resultList.get(i).setModelClassificationStr(mapOfDictData.get(Integer.parseInt(resultList.get(i).getModelClassification())));
            }
            return resultList;
        });
    }

第二个删除，要注意一下被引用的不能删

@Override
public int deleteById(String id) {
        if(id==null) return 0;
        //要看一下是否被风险场景引用，被引用则不能删
        if(rcSceneDtlMapper.countOfRiskModel(Long.parseLong(id))==0){
            rcModelDtlMapper.deleteByModelId(Long.parseLong(id));
            this.mapper.deleteById(Long.parseLong(id));
            return 1;
        }
        return 0;
    }

第四个更新模型，如果前端传的有数据错误，则不予更新。当风险项目没有变化，则不用更新

@Override
@Transactional
public Integer updateRow(RcModelDomain modelDomain, List<Long> rcRiskDefIds) {
    RcModelDomain rc1=rcModelMapper.selectById(modelDomain.getModelId());
    if(rc1==null) return 0;
    rc1.setUpdateTime(new Date());
    rc1.setUpdateBy(LoginUser.getUserName());
    if(StringUtils.isNotBlank(modelDomain.getModelName())){
        rc1.setModelName(modelDomain.getModelName());
    }else{
        return -1;
    }
    if(StringUtils.isNotBlank(modelDomain.getModelClassification())){
        rc1.setModelClassification(modelDomain.getModelClassification());
    }else{
        return -1;
    }
    //因为是写数据库，所以不需要写入ModelClassificationStr
    rcModelMapper.updateRow(rc1);
    List<Long> dataBaseRiskDefIds=rcModelDtlMapper.selectAllRiskDefByModelId(rc1.getModelId());
    if(dataBaseRiskDefIds==null || rcRiskDefIds==null) return -1;
    if(CollectionUtils.isEqualCollection(rcRiskDefIds,dataBaseRiskDefIds)){
        //两个列表相等，风险项目没有修改，则不用修改数据库
        return 1;
    }else{
        rcModelDtlMapper.deleteByModelId(rc1.getModelId());
        rcModelDtlMapper.insertDtlBatch(rc1.getModelId(),rcRiskDefIds);
    }
    return 1;
}

    @Override
    public GeneratorReportVO getModelsAndRisksByScene(Long sceneId) {
        GeneratorReportVO vo1=new GeneratorReportVO();
        List<Long> modelIdList=rcSceneDtlMapper.selectAllModelsBySceneId(sceneId);
        List<RcModelDomain> modelList=rcModelMapper.selectBatchIds(modelIdList);
        List<GeneratorReportVO.RcModel> rcModelList=new ArrayList<>();

        HashMap<Integer,String> mapOfDictData=new HashMap<>();
        List<DictDataDomain> modelStatusList = GlobalDictUtils.getDataList("modelStatus");
        for(int i=0;i<modelStatusList.size();i++){
            mapOfDictData.put(Integer.parseInt(modelStatusList.get(i).getDictValue()),modelStatusList.get(i).getDictLabel());
        }
        // 存储的为  1 -> 模型A     2->模型B


        for(RcModelDomain rc1:modelList){
            GeneratorReportVO.RcModel rcModel1=new GeneratorReportVO.RcModel();
            BeanUtils.copyProperties(rc1,rcModel1);
            rcModel1.setName(rcModel1.getModelName());
            rcModel1.setClassificationStr(mapOfDictData.get(Integer.parseInt(rc1.getModelClassification())));
            rcModel1.setId(rcModel1.getModelId().toString());
            rcModelList.add(rcModel1);
        }
        vo1.setRcModelList(rcModelList);

        //将一些字段从数字转为中文
        HashMap<Integer,String> mapOfDictDataRisk=new HashMap<>();
        List<DictDataDomain> modelStatusListRisk = GlobalDictUtils.getDataList("riskCategory");
        for(int i=0;i<modelStatusListRisk.size();i++){
            mapOfDictDataRisk.put(Integer.parseInt(modelStatusListRisk.get(i).getDictValue()),modelStatusListRisk.get(i).getDictLabel());
        }
        // 存储的为  1 -> 风险类别A     2->风险类别B    3->风险类别C
        for(GeneratorReportVO.RcModel rc1:rcModelList){
            List<Long> riskDefIdList=rcModelDtlMapper.selectAllRiskDefByModelId(rc1.getModelId());
            List<RcRiskDefDomain> riskDefList=rcRiskDefMapper.selectBatchIds(riskDefIdList);
            List<GeneratorReportVO.RiskInfo> temp2=new ArrayList<>();
            for (RcRiskDefDomain rcRiskDefDomain : riskDefList) {
                GeneratorReportVO.RiskInfo temp1=new GeneratorReportVO.RiskInfo();
                BeanUtils.copyProperties(rcRiskDefDomain, temp1);
                temp1.setName(temp1.getRiskName());
                temp1.setClassificationStr(mapOfDictDataRisk.get(Integer.parseInt(rcRiskDefDomain.getRiskCategory())));
                temp1.setId(rc1.getId()+"_"+temp1.getRiskId().toString());
                temp2.add(temp1);
            }
            rc1.setRiskDefList(temp2);
        }
        return vo1;
    }
